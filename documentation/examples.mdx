# Examples

## String

```js
import { pickitt } from 'pickitt'

const arrayOfStrings = ['item1', 'item2', 'item3']

const randomElement = pickitt(arrayOfStrings) // <-- 'item1', 'item2', or 'item3'
```

## Number

```js
import { pickitt } from 'pickitt'

const arrayOfNumbers = [1, 2, 3]

const randomElement = pickitt(arrayOfNumbers) // <-- 1, 2, or 3
```

## Boolean

```js
import { pickitt } from 'pickitt'

const arrayOfBooleans = [true, false]

const randomElement = pickitt(arrayOfBooleans) // <-- true, or false
```

## Empty Array

```js
import { pickitt } from 'pickitt'

const emptyArray = []

const randomElement = pickitt(emptyArray) // <-- undefined
```

## Array of Arrays

```js
import { pickitt } from 'pickitt'

const arrayOfStrings = ['item1', 'item2', 'item3']
const arrayOfNumbers = [1, 2, 3]
const arrayOfBooleans = [true, false]

const arrayOfArrays = [arrayOfStrings, arrayOfNumbers, arrayOfBooleans]

const randomElement = pickitt(arrayOfArrays) // <-- ['item1', 'item2', 'item3'], [1, 2, 3], or [true, false]
```
